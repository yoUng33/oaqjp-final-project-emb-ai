"""
This module provides a Flask server for an Emotion Detection application.
It includes routes for rendering the index page and processing emotion analysis.
"""
from flask import Flask, render_template, request
from EmotionDetection.emotion_detection import emotion_detector

app = Flask("Emotion Detector")

@app.route("/")
def render_index_page():
    """
    Renders the main index page of the application.
    """
    return render_template('index.html')

@app.route("/emotionDetector")
def send_emotion():
    """
    Retrieves text from request arguments, processes it for emotion detection,
    and returns a formatted string result.
    """
    # Retrieve the text to analyze from the request arguments
    text_to_analyze = request.args.get('textToAnalyze')

    # Use the imported emotion_detector function
    response = emotion_detector(text_to_analyze)

    # Extract the dominant emotion (removes it from the dict)
    dominant = response.pop("dominant_emotion")

    if dominant is None:
        return "Invalid text! Please try again!"

    # Format the remaining items (anger, disgust, etc.)
    parts = [f"'{k}': {v}" for k, v in response.items()]

    # Create the grammar-friendly list: "a, b, and c"
    formatted_scores = ", ".join(parts[:-1]) + f" and {parts[-1]}"

    # Construct the final string
    output = (f"For the given statement, the system response is {formatted_scores}. "
              f"The dominant emotion is {dominant}.")

    return output

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
